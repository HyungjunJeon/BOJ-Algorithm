'use strict';
const readline = require('readline');
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

let input = [];

rl.on('line', line => {
    input.push(line);
  })
  .on('close', () => {
    for(let i = 1; i <= input[0]; i++) {
      input[i] % 2 === 0 ? console.log(`${input[i]} is even`) : console.log(`${input[i]} is odd`);
    }
    process.exit();
  });
