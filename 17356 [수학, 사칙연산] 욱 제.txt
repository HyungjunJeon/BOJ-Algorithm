'use strict';
const readline = require('readline');
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

let input = [];

rl.on('line', line => {
    input = line.split(' ').map((el) => parseInt(el));
  })
  .on('close', () => {
    let M = (input[1] - input[0]) / 400;
    let answer = 1 / (1 + Math.pow(10, M));
    console.log(answer);
    process.exit();
  });
