'use strict';
const readline = require('readline');
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

let input = [];

rl.on('line', line => {
    input.push(line.split(' ').map((el) => parseInt(el)));
  })
  .on('close', () => {
    let answer;

    for(let i =1; i <= input[0][0]; i++) {
      answer = Math.ceil(input[i][0] / input[i][1]);
      console.log(answer);
    }
    process.exit();
  });
