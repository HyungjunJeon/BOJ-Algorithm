'use strict';
const readline = require('readline');
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

let input = [];

rl.on('line', line => {
    input = line.split(' ').map((el) => parseInt(el));
  })
  .on('close', () => {
    let sum = 0;

    for(let i = 0; i < input.length; i++) {
      sum += input[i];
    }

    console.log(sum);
    process.exit();
  });
