'use strict';
const readline = require('readline');
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

let input = [];

rl.on('line', line => {
    input.push(line.split(' ').map((el) => parseInt(el)));
  })
  .on('close', () => {
    let count = 0;
    for(let i = 0; i < input[1].length; i++) {
      if(input[0][0] === input[1][i]) {
        count++;
      }
    }
    console.log(count);
    process.exit();
  });
