'use strict';
const readline = require('readline');
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

let input = [];

rl.on('line', function (line) {
    input.push(line);
  })
  .on('close', function () {
    for(let i = input[0]; i >= 1; i--) {
      console.log(i);
    }
    process.exit();
  });
